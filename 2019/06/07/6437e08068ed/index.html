Unity Editor 工具脚本速查

2019-06-07 1.1k 字
Unity C#

MenuItem — 菜单项入口

// 静态方法 + MenuItem 特性 → 自动出现在菜单栏
[MenuItem("Tools/Do Something")]
public static void DoSomething()
{
    Debug.Log("菜单点击");
}

// 带快捷键（% = Ctrl/Cmd, # = Shift, & = Alt, _ = 无修饰符）
[MenuItem("Tools/Batch Rename %#r")]  // Ctrl+Shift+R
public static void BatchRename() { }

// 带验证函数（返回 bool 控制菜单是否可点击）
[MenuItem("Tools/Conditional Action", true)]
public static bool ValidateAction()
{
    return Selection.activeGameObject != null;
}

要点：验证函数名与执行函数名相同，第二个参数 true 标记为验证函数。

Selection — 获取当前选中对象

// 当前选中的 GameObject
var go = Selection.activeGameObject;

// 当前选中的 Transform
var t = Selection.activeTransform;

// 当前选中的对象（Component、Asset 等）
var obj = Selection.activeObject;

// 多选
var objects = Selection.objects;       // Object[]
var gameObjects = Selection.gameObjects; // GameObject[]
var transforms = Selection.transforms; // Transform[]

EditorWindow — 自定义工具面板

public class MyToolWindow : EditorWindow
{
    [MenuItem("Tools/My Tool Window")]
    public static void ShowWindow()
    {
        GetWindow<MyToolWindow>("我的工具");
    }

    void OnGUI()
    {
        GUILayout.Label("自定义工具面板", EditorStyles.boldLabel);

        if (GUILayout.Button("执行操作"))
        {
            DoWork();
        }
    }

    void DoWork()
    {
        foreach (var go in Selection.gameObjects)
        {
            go.name = go.name.Replace("(Clone)", "");
        }
    }
}

CustomEditor — 自定义 Inspector

[CustomEditor(typeof(EnemyData))]
public class EnemyDataEditor : Editor
{
    public override void OnInspectorGUI()
    {
        // 绘制默认 Inspector
        DrawDefaultInspector();

        // 追加自定义按钮
        var data = (EnemyData)target;
        if (GUILayout.Button("随机生成属性"))
        {
            data.hp = Random.Range(50, 200);
            data.attack = Random.Range(10, 50);
            EditorUtility.SetDirty(data);
        }
    }
}

关键 API：

  • target — 当前编辑的对象（object 类型，需强转）。
  • serializedObject — 序列化对象，用于操作 SerializedProperty。
  • EditorUtility.SetDirty(obj) — 标记对象已修改（支持 Undo）。
  • Undo.RecordObject(obj, "操作描述") — 注册到撤销栈。

PropertyDrawer — 自定义字段绘制

[Serializable]
public struct MinMaxRange
{
    public float min;
    public float max;
}

[CustomPropertyDrawer(typeof(MinMaxRange))]
public class MinMaxRangeDrawer : PropertyDrawer
{
    public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
    {
        var minProp = property.FindPropertyRelative("min");
        var maxProp = property.FindPropertyRelative("max");

        EditorGUI.BeginProperty(position, label, property);

        // 双滑块
        float min = minProp.floatValue;
        float max = maxProp.floatValue;

        EditorGUI.MinMaxSlider(position, label, ref min, ref max, 0f, 100f);

        minProp.floatValue = min;
        maxProp.floatValue = max;

        EditorGUI.EndProperty();
    }
}

实用案例

案例 1：批量修改 Prefab 的 Layer

[MenuItem("Tools/Set Selected to Layer 'Environment'")]
public static void SetToEnvironmentLayer()
{
    foreach (var go in Selection.gameObjects)
    {
        SetLayerRecursive(go, LayerMask.NameToLayer("Environment"));
    }
}

static void SetLayerRecursive(GameObject go, int layer)
{
    go.layer = layer;
    foreach (Transform child in go.transform)
        SetLayerRecursive(child.gameObject, layer);
}

案例 2：查找场景中 Missing 脚本的物体

[MenuItem("Tools/Find Missing Scripts")]
public static void FindMissingScripts()
{
    var allObjects = Resources.FindObjectsOfTypeAll<GameObject>();
    int count = 0;

    foreach (var go in allObjects)
    {
        var components = go.GetComponents<Component>();
        for (int i = 0; i < components.Length; i++)
        {
            if (components[i] == null)
            {
                Debug.Log($"Missing 脚本: {go.name}", go);
                count++;
            }
        }
    }

    Debug.Log($"找到 {count} 个 Missing 脚本");
}

案例 3：一键生成预制体缩略图

[MenuItem("Assets/Capture Prefab Thumbnail")]
public static void CaptureThumbnail()
{
    var go = Selection.activeGameObject;
    var rt = new RenderTexture(256, 256, 16);
    var cameraObj = new GameObject("ThumbCam", typeof(Camera));
    var cam = cameraObj.GetComponent<Camera>();
    cam.targetTexture = rt;
    cam.Render();

    RenderTexture.active = rt;
    var tex = new Texture2D(256, 256, TextureFormat.RGBA32, false);
    tex.ReadPixels(new Rect(0, 0, 256, 256), 0, 0);
    tex.Apply();

    var bytes = tex.EncodeToPNG();
    var path = AssetDatabase.GetAssetPath(go);
    var dir = System.IO.Path.GetDirectoryName(path);
    System.IO.File.WriteAllBytes($"{dir}/{go.name}_thumb.png", bytes);

    Object.DestroyImmediate(cameraObj);
    RenderTexture.active = null;
    AssetDatabase.Refresh();
}
